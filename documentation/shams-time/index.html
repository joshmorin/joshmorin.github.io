Crate shams_time [−] [src]

Islamic prayer times and other time events that occur throughout a given day.

Example

    use shams_time::ShamsTimeSpec;
     
    // Specify location and calculation details
    let times = ShamsTimeSpec::new().method(6)
                                   .coordinates(29.366667, 47.966667)
                                   .timezone(3.0).temperature(25.0)
                                   .pressure(1010.0).altitude(20.0)
                                   .today().calculate();
     
    // Get the midpoint between sunset and sunrise
    let (hours, minutes, _) = times.midnight().expect("unsupported location");

    // Get Al-fajr prayer time
    let (fajr_hours, _, _, _) = times.fajr();
     
    println!("True midnight at {}:{:02}; {} hours after fajr", 
        hours, minutes, hours - fajr_hours);
