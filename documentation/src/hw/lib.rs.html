macro_rules! log {($($arg:tt)*)=>({use std::env; if env::var("HWLIB_ERROR_LOG").is_ok() {
        print!("[HWLIB] {}:{}: ", file!(), line!());println!($($arg)*)}})}


macro_rules! log_n_ret {
    ($gerund_phrase:expr, $($arg:tt)*) => ({
        #[allow(unused_imports)]
        use ::error::{Result, Error};
        
        let error = Error::failure($gerund_phrase, &format!($($arg)*));;
        log!("{}", error);
        return Err(error);
    })
}

#[cfg(target_os="linux")] mod linux;
#[cfg(target_os="linux")] pub use linux::{privileged, unprivileged};







macro_rules! try_privileged_then_unprivileged {
    ($device: ident, $func: ident) => ({
        match privileged::$device::$func() {
            Ok(o) => Ok(o),
            _ => {
                unprivileged::$device::$func()
            }
        }
    });
    ($device: ident, $func: ident, $arg1: expr) => ({
        match privileged::$device::$func($arg1) {
            Ok(o) => Ok(o),
            _ => unprivileged::$device::$func($arg1)
        }
    });
    
}

/// System Handle
#[derive(Debug)]
pub struct System;
impl System {
    /// Immediate system power off.
    /// 
    /// **Warning:** May cause data loss and other undesirable side-effects associated with
    /// an "Unclean" shutdown. Consider `System::shutdown` if this is 
    /// undesirable.
    pub fn power_off() -> Result<()> {try_privileged_then_unprivileged!(System, power_off)}
    /// Immediate system reboot.
    /// 
    /// **Warning:** May cause data loss and other undesirable side-effects associated with
    /// an "Unclean" shutdown. Consider `System::restart` if this is 
    /// undesirable.
    pub fn reboot() -> Result<()> {try_privileged_then_unprivileged!(System, reboot)}
    /// Sends the shutdown signal for an operating system-assisted power off.
    ///
    /// Exact behavior is system-dependent, and may entail additional user action 
    /// (confirmation). Successful result may not imply successful shutdown 
    /// initiation.
    pub fn shutdown() -> Result<()> {try_privileged_then_unprivileged!(System, shutdown)}
    /// Sends the restart signal for an operating system-assisted reboot.
    ///
    /// Exact behavior is system-dependent, and may entail additional user action 
    /// (confirmation). Successful result may not imply successful shutdown 
    /// initiation.
    pub fn restart() -> Result<()> {try_privileged_then_unprivileged!(System, restart)}
    /*
    /// Sends the sleep signal for an operating system-assisted reboot.

    /// (confirmation). Successful result may not imply successful shutdown

    pub fn sleep() -> Result<()> {try_privileged_then_unprivileged!(System, suspend)}
    */
    
}
/// System Audio Output Handle
#[derive(Debug)]
pub struct Speaker;
impl Speaker {
    /// Sets system audio output volume to a specified level. `level` range 
    /// must be a value either 0.0 (min), 1.0 (max), or any value in between. 
    pub fn volume(level: f64) -> Result<()> {try_privileged_then_unprivileged!(Speaker, volume, level)}
    /// Mutes system audio output
    pub fn mute() -> Result<()> {try_privileged_then_unprivileged!(Speaker, mute)}
    /// Unmutes system audio output
    pub fn unmute() -> Result<()> {try_privileged_then_unprivileged!(Speaker, unmute)}
}

/// System Video Output Handle
#[derive(Debug)]
pub struct Display;
impl Display {
    /// Sets system video output brightness to a specified intensity. 
    /// `intensity` range must be a value either 0.0 (min), 1.0 (max), 
    /// or any value in between. 
    pub fn brightness(intensity: f64) -> Result<()> {try_privileged_then_unprivileged!(Display, brightness, intensity)}
    /// Turns off video output for the system.
    pub fn off() -> Result<()> {try_privileged_then_unprivileged!(Display, off)}
    /// Turns on video output for the system.
    pub fn on() -> Result<()> {try_privileged_then_unprivileged!(Display, on)}
}

fn it_works() {
    use std::{env, thread};
    macro_rules! test_gradient {
        ($device: ident, $func: ident) => ({
            println!("testing gradient for {}::{}...", stringify!($device), stringify!($func));
            let resolution = 200.0;
            //let mut g = Hw::new().$f().unwrap_or_else(|e| panic!("{}", e));
            let time = 0.5; //seconds
            for i in 0..(resolution as u32 +1) { 
                if $device::$func(((i as f64-(resolution/2.0)).abs())/100.0).is_err() {
                    break;
                }
                thread::sleep_ms((time*1000.0/resolution) as u32);
            }
        })
    }
    
    macro_rules! test_toggle {
        ($device: ident, $off: ident, $on: ident) => ({
            println!("toggling test for {}::{}/{}...", stringify!($device), stringify!($on), stringify!($off));
            //let mut g = Hw::new().$f().unwrap_or_else(|e| panic!("{}", e));
            loop {
                println!("{} {}", stringify!($off), stringify!($device));
                if $device::$off().is_err() {break};
                thread::sleep_ms(1000);
                println!("{} {}", stringify!($on), stringify!($device));
                if $device::$on().is_err() {break};
                break;
            }
        })
    }
    
    env::set_var("HWLIB_ERROR_LOG", "Y");
    

    test_gradient!(Display, brightness);
    test_gradient!(Speaker, volume);
    test_toggle!(Display, off, on);
