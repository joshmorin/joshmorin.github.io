Longest day of the year

extern crate shams_time;
extern crate time;
 
fn main() {
    use shams_time::ShamsTimeSpec;
    use time::Duration;
    use std::default::Default;
     
    // Get an iterator that yields a ShamsTime object for every day of the 
    // year 2003
    let year = ShamsTimeSpec::city("Makkah").expect("unknown city").year(2003);
     
    // iterate of all days and search for longest day
    let (date, duration) = year.fold((Default::default(), 0), |max, day| {
        let daytime = day.daytime().expect("no sunset or sunrise");
        if daytime > max.1 {
            (day.date(), daytime)
        } 
        else {
            max
        }
    });

    // converting UNIX time to hours, minutes, and seconds
    let duration = Duration::seconds(duration);
    println!(concat!("longest day in Makkah for the year {} was on {}/{}, ", 
                     "lasting for {} hours."), 
                      date.year, date.day, date.month,
                      duration.num_hours());
}
