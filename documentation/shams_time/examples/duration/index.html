The time between Al-Fajr and true midnight

extern crate shams_time;
extern crate time;
 
fn main() {
    use shams_time::ShamsTimeSpec;
    use time::Duration;

    // Specify location and calculation details
    let st = ShamsTimeSpec::new().method(6)
                                 .coordinates(29.366667, 47.966667)
                                 .timezone(3.0).temperature(25.0)
                                 .pressure(1010.0).elevation(20.0)
                                 .today().calculate();
     
    // Get true midnight and Al-Fajr
    let midnight = st.midnight().expect("midnight sun?");
    let fajr = st.fajr().0.expect("unsupported location");
     
    println!("True midnight is {} hours after Al-Fajr", 
        Duration::seconds(midnight - fajr).num_hours());

}
