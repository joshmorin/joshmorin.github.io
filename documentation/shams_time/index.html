Islamic prayer times and other time events that occur throughout a given day.

Examples: The time between Al-Fajr and true midnight

extern crate shams_time;
extern crate time;
 
fn main() {
    use shams_time::ShamsTimeSpec;
    use time::Duration;

    // Specify location and calculation details
    let st = ShamsTimeSpec::new().method(6)
                                 .coordinates(29.366667, 47.966667)
                                 .timezone(3.0).temperature(25.0)
                                 .pressure(1010.0).altitude(20.0)
                                 .today().calculate();
     
    // Get true midnight and Al-fajr
    let midnight = st.midnight().expect("midnight sun?");
    let fajr = st.fajr().0.expect("unsupported location");
     
    println!("True midnight is {} hours after fajr", 
        Duration::seconds(midnight - fajr).num_hours());

}

Example: Longest day of the year

extern crate shams_time;
extern crate time;
 
fn main() {
    use shams_time::ShamsTimeSpec;
    use time::Duration;
    use std::default::Default;
     
    // Get an iterator that yeilds a ShamsTime object for every day of the 
    // year 2003
    let year = ShamsTimeSpec::city("Makkah").expect("unknown city").year(2003);
     
    // iterate of all days and search for longest day
    let (date, durration) = year.fold((Default::default(), 0), |max, day| {
        let daytime = day.daytime().expect("no sunset or sunrise");
        if daytime > max.1 {
            (day.date(), daytime)
        } 
        else {
            max
        }
    });

    // converting unix time to hours, minutes, and seconds
    let durration = Duration::seconds(durration);
    println!(concat!("longest day in Makkah for the year {} was on {}/{}, ", 
                     "lasting for {} hours."), 
                      date.year, date.day, date.month,
                      durration.num_hours());
}
